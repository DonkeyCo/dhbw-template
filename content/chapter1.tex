\section{Document Structure}
\label{sec:structure}

Use these commands for document hierarchy:
\begin{itemize}
    \item \code{\textbackslash chapter\{Title\}} - Top-level divisions
    \item \code{\textbackslash section\{Title\}} - Main sections
    \item \code{\textbackslash subsection\{Title\}} - Subdivisions
\end{itemize}

\subsection{Text Formatting}

Basic formatting options:
\begin{itemize}
    \item \textbf{Bold text} - \code{\textbackslash textbf\{text\}}
    \item \textit{Italic text} - \code{\textbackslash textit\{text\}}
    \item \texttt{Monospace} - \code{\textbackslash texttt\{text\}}
    \item \important{Important} - \code{\textbackslash important\{text\}}
\end{itemize}

\subsection{Code Blocks}
\begin{lstlisting}[language=JavaScript, caption=Example function, label=lst:example-js]
function calculateSum(numbers) {
    let sum = 0;
    for (let i = 0; i < numbers.length; i++) {
        sum += numbers[i];
    }
    return sum;
}

const result = calculateSum([1, 2, 3, 4, 5]);
console.log("Sum:", result);
\end{lstlisting}
